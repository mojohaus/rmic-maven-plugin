9    * Represents a single invocation of the rmic compiler.
10   */
11  class Invocation
12  {
13      private List<String> arguments;
14  
15      Invocation( String[] arguments )
16      {
17          this.arguments = Arrays.asList( arguments );
18      }
19  
20      @SuppressWarnings( "unchecked" )
21      List<String> getArguments()
22      {
23          return Collections.unmodifiableList( arguments );
24      }
25  
26      public String toString()
27      {

31      }
32  }
